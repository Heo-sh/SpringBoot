CREATE SEQUENCE SEQ_PRODUCT;

CREATE TABLE PRODUCT(
	PRODUCT_ID NUMBER PRIMARY KEY,		--제품번호
	PRODUCT_NAME VARCHAR2(500) NOT NULL,	--제품명
	PRODUCT_STOCK NUMBER DEFAULT 0,	--제품재고량
	PRODUCT_PRICE NUMBER DEFAULT 0,		--제품가격
	REGISTER_DATE DATE DEFAULT SYSDATE,	--제품등록날짜
	UPDATE_DATE DATE DEFAULT SYSDATE	--제품등록날짜를 업데이트한 날짜
);
